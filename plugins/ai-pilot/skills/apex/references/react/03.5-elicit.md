## Example: Component Review

```tsx
// Before elicitation
function UserList({ users }) {
  return (
    <div>
      {users.map(u => <div>{u.name}</div>)}
    </div>
  )
}

// After UX-01 + TEST-01 + ARCH-01
interface UserListProps {
  users: User[]
}

function UserList({ users }: UserListProps) {
  if (users.length === 0) {
    return <p role="status">No users found</p>
  }

  return (
    <ul aria-label="User list">
      {users.map(user => (
        <li key={user.id}>{user.name}</li>
      ))}
    </ul>
  )
}
```
